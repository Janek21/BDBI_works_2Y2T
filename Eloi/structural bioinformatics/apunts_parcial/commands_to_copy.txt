#After we got file.fasta we FIRST put the target.fa into new file named pssm.fasta
cat target.fa > pssm.fasta

#And SECONDLY we also introduce the file.fasta into the same filecat file.fasta >> pssm.fasta

#Finally we perform clustalw2 to get ".aln" file
clustalw2 pssm.fasta

#We convert ".aln" to ".fa" 
perl ~/Desktop/perl_scripts/aconvertMod2.pl -in c -out f <pssm.aln>pssm.fa

#We use ".fa" in psiblast as following to get results for our homemade pssm.
psiblast -in_msa pssm.fa -out target_pdb_specific.out -db ~/Desktop/databases/pdb_seq
